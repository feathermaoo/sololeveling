    <script>

        // --- 遊戲核心數據資料 ---

        let gameState = {

            level: 1,

            xp: 0,

            gold: 0,

            inventory: [],

            tasks: [],

            books: []

        };


        // --- 1-100級 階梯式經驗值引擎 ---

        function getRequiredXp(level) {

            if (level === 1) return 20;

            if (level >= 2 && level < 5) return 40;

            if (level >= 5 && level < 15) return 60;

            if (level >= 15 && level < 40) return 65 + Math.round((level - 15) * 2.3); 

            if (level >= 40 && level < 65) return 122 + Math.round((level - 40) * 2.4);

            if (level >= 65 && level < 80) return 185 + Math.round((level - 65) * 4.6);

            if (level >= 80 && level < 95) return 260 + (level - 80) * 10;

            if (level === 95) return 450;

            if (level === 96) return 500;

            if (level === 97) return 600;

            if (level === 98) return 700;

            if (level === 99) return 800;

            return 999999; // 滿等

        }


        // --- 假日檢測 (週六日 1.5 倍加成) ---

        function getRewardMultiplier() {

            const day = new Date().getDay();

            return (day === 0 || day === 6) ? 1.5 : 1;

        }


        // --- 獲得獎勵機制 (核心邏輯) ---

        function gainRewards(baseXp, baseGold) {

            const mult = getRewardMultiplier();

            let finalXp = Math.round(baseXp * mult);

            let finalGold = Math.round(baseGold * mult);


            gameState.xp += finalXp;

            gameState.gold += finalGold;


            // 檢查升級

            while (gameState.xp >= getRequiredXp(gameState.level) && gameState.level < 100) {

                gameState.xp -= getRequiredXp(gameState.level);

                gameState.level++;

                alert(`🎉 恭喜羽毛升級到 Lv.${gameState.level}！額外獲得滿血金幣獎勵 50 🪙`);

                gameState.gold += 50;

            }


            saveAndRender();

        }


        // --- 每日正向回饋 ---

        function addGoodDeed() {

            const input = document.getElementById('input-good');

            if (!input.value.trim()) return;

            

            const ul = document.getElementById('list-goods');

            const li = document.createElement('li');

            li.innerHTML = `<span>✨ ${input.value}</span> <span style="color:green;">+10 XP</span>`;

            ul.appendChild(li);

            

            input.value = '';

            gainRewards(10, 5);

        }


        // --- 24小時任務管理 ---

        function addTask() {

            const input = document.getElementById('input-task');

            if (!input.value.trim()) return;


            const task = {

                id: Date.now(),

                text: input.value,

                createdTime: new Date().toLocaleString(),

                done: false

            };

            gameState.tasks.push(task);

            input.value = '';

            saveAndRender();

        }


        function completeTask(id) {

            gameState.tasks = gameState.tasks.filter(t => t.id !== id);

            // 24小時內完成，給予優渥獎勵

            gainRewards(20, 15);

        }


        // --- 讀書進度管理 ---

        function createBook() {

            const name = document.getElementById('book-name').value;

            const total = parseInt(document.getElementById('book-total').value);

            if (!name || !total) return;


            gameState.books.push({

                id: Date.now(),

                name: name,

                total: total,

                current: 0

            });

            document.getElementById('book-name').value = '';

            document.getElementById('book-total').value = '';

            saveAndRender();

        }


        function updateBookProgress(id, newPage) {

            const book = gameState.books.find(b => b.id === id);

            if (!book) return;

            

            newPage = Math.min(book.total, Math.max(0, parseInt(newPage) || 0));

            const oldPercent = Math.floor((book.current / book.total) * 100);

            const newPercent = Math.floor((newPage / book.total) * 100);


            if (newPercent > oldPercent) {

                // 每提升 1% 給 5 XP / 2 金幣

                const percentDiff = newPercent - oldPercent;

                gainRewards(percentDiff * 5, percentDiff * 2);

                

                if (newPercent === 100) {

                    alert(`📖 精通大成功！完讀《${book.name}》，額外大獎 100 XP / 50 金幣！`);

                    gainRewards(100, 50);

                }

            }

            

            book.current = newPage;

            if(book.current === book.total) {

                gameState.books = gameState.books.filter(b => b.id !== id); // 讀完移出

            }

            saveAndRender();

        }


        // --- 商店與裝潢功能 ---

        const spriteMap = { dog: '🐕', cat: '🐈', tree: '🌳', radio: '📻' };

        

        function buyItem(type, cost) {

            if (gameState.gold < cost) {

                alert("🪙 金幣不足！快去解任務賺錢吧！");

                return;

            }

            gameState.gold -= cost;

            gameState.inventory.push(type);

            spawnSprite(type);

            saveAndRender();

        }


        function spawnSprite(type) {

            const farm = document.getElementById('farm');

            const sprite = document.createElement('div');

            sprite.className = 'pixel-sprite';

            sprite.innerText = spriteMap[type];

            

            // 隨機擺放位置

            sprite.style.left = Math.random() * (farm.clientWidth - 40) + 'px';

            sprite.style.top = Math.random() * (farm.clientHeight - 60) + 40 + 'px';


            // 點擊互動機制

            sprite.onclick = (e) => {

                e.stopPropagation();

                const bubble = document.getElementById('bubble');

                bubble.style.left = sprite.style.left;

                bubble.style.top = (parseFloat(sprite.style.top) - 30) + 'px';

                bubble.style.display = 'block';


                if (type === 'dog') {

                    const quotes = ["羽毛今天超棒！", "主人，今天也要一起升級！", "汪！(對你搖尾巴)", "休息一下，摸摸我吧！"];

                    bubble.innerText = quotes[Math.floor(Math.random() * quotes.length)];

                } else if (type === 'cat') {

                    bubble.innerText = "喵~ (在曬太陽不理你)";

                } else if (type === 'radio') {

                    bubble.innerText = "🎵 正在播放：像素鄉村 Lo-Fi 輕音樂... 🎵";

                } else {

                    bubble.innerText = "🍎 果樹正在茁壯成長！";

                }

                

                setTimeout(() => bubble.style.display = 'none', 2500);

            };


            farm.appendChild(sprite);

        }


        // --- 畫面渲染與資料儲存 ---

        function saveAndRender() {

            localStorage.setItem('solo_leveling_farm', JSON.stringify(gameState));


            // 更新狀態

            document.getElementById('lbl-level').innerText = gameState.level;

            document.getElementById('lbl-xp').innerText = gameState.xp;

            const nextXp = getRequiredXp(gameState.level);

            document.getElementById('lbl-next-xp').innerText = nextXp;

            document.getElementById('lbl-gold').innerText = gameState.gold;


            const percent = (gameState.xp / nextXp) * 100;

            document.getElementById('xp-bar').style.width = `${percent}%`;


            // 渲染假日標籤

            const mult = getRewardMultiplier();

            document.getElementById('weekend-alert').innerHTML = mult > 1 

                ? `<span class="weekend-badge">📅 週末狂歡 Buff 啟動中：所有收益 1.5 倍！</span>` : '';


            // 渲染 24H 任務

            const taskList = document.getElementById('list-tasks');

            taskList.innerHTML = '';

            gameState.tasks.forEach(t => {

                taskList.innerHTML += `

                    <li>

                        <div>

                            <strong>⚔️ ${t.text}</strong><br>

                            <small style="color:#777;">發布時間: ${t.createdTime}</small>

                        </div>

                        <button onclick="completeTask(${t.id})">回報完成</button>

                    </li>

                `;

            });


            // 渲染書籍

            const bookSection = document.getElementById('book-section');

            bookSection.innerHTML = '';

            gameState.books.forEach(b => {

                const p = Math.floor((b.current / b.total) * 100);

                bookSection.innerHTML += `

                    <div style="background:#fff; border:2px solid var(--border-color); padding:10px; margin-top:5px;">

                        <strong>📖 ${b.name}</strong> (${p}%)

                        <div class="bar-container" style="width:100%; height:10px; margin:5px 0;">

                            <div class="bar-fill" style="width:${p}%; background:var(--primary-color);"></div>

                        </div>

                        <div style="display:flex; justify-content:space-between; align-items:center;">

                            <span>進度：<input type="number" value="${b.current}" style="width:60px; padding:2px;" 

                                onchange="updateBookProgress(${b.id}, this.value)"> / ${b.total} 頁</span>

                        </div>

                    </div>

                `;

            });

        }


        // --- 初始化讀取 ---

        window.onload = function() {

            const saved = localStorage.getItem('solo_leveling_farm');

            if (saved) {

                gameState = JSON.parse(saved);

                // 重新生成農舍裡原本就有的物件

                gameState.inventory.forEach(type => spawnSprite(type));

            }

            saveAndRender();

        };
